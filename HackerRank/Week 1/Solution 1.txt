def plusMinus(arr):
    # Write your code here
    n = len(arr)
    positive_count = sum(1 for num in arr if num > 0)
    negative_count = sum(1 for num in arr if num < 0)
    zero_count = sum(1 for num in arr if num ==0)
    
    print(f"{positive_count / n:.6f}")
    print(f"{negative_count / n:.6f}")
    print(f"{zero_count / n:.6f}")

if __name__ == '__main__':
    n = int(input().strip())

    arr = list(map(int, input().rstrip().split()))

    plusMinus(arr)
